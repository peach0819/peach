48预售提货卡转实货支付金额  v3新增

具体表逻辑参考：

SELECT order_id,
           sum(CASE WHEN card_serial_cause in (2) and template_card_id  in (1641，1642，1643，1644，1645，1646，1647，1648，1649)  THEN card_fund_serial_amt ELSE 0 END) AS pickup_card_amount,--预售提货卡支付金额
           sum(CASE WHEN card_serial_cause in (4) and template_card_id  in (1641，1642，1643，1644，1645，1646，1647，1648，1649)  THEN card_fund_serial_amt ELSE 0 END) AS pickup_card_amount1--预售提货卡退款金额
    WHERE dayid = '${v_date}'
    AND hi_card_type = 1 -- 提货卡
    AND card_serial_direction = 0 -- 支出
    AND card_serial_cause in(2,4) -- 支付支出
        -- 理论上是可以加上时间为当月做过滤
    AND order_id IS NOT NULL
    GROUP BY order_id
