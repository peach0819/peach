-- 分类1个人薪资计算-非提货卡
SELECT
  sum(commission_actual_amount)
from
  (
    select
      *
    from
      ytdw.dw_salary_base_data_d
    WHERE
      item_style_name = 'B'
      and is_split = 'BD'
      and is_pickup_recharge_order_name = '否'
      and service_user_id = '0035c5a0ac7a4b71aa62239db281d3e6'
      and dayid = '$v_date'
  ) a
 left join 
  ( SELECT id,name
     FROM dwd_brand_d
     where dayid = '$v_date'
     and (array_contains(split(tags,','),'42')  or  array_contains(split(tags,','),'41'))
   ) celeron on celeron.name=a.brand_name
  LEFT JOIN 
( select brand_name from dw_offline_spec_brand_d  where dayid ='$v_date' group by brand_name
   ) spec_brand on spec_brand.brand_name=a.brand_name
where spec_brand.brand_name is null and ( celeron.name is not null and a.category_id_first=10)

or a.category_id_first =7999 
or a.category_id_second in (2807,2721,2712,6820) 
or a.brand_id = 13818
