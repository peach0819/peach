1、select语句中，表自带库名、时间变量替换

通过com.alibaba.druid解析器，解析所有表，并添加上yt_crm或ytdw的库名；解析所有ON、WHERE语句，把$v_date替换为${v_date}

            "yt_date_format",
            "getValueFromKVString",
            "sort_connect",
            "get_service_info",
            "get_role",
            "convertRedpillCode",

    //迁移的业务db
    private String dbName;

    private String sql;

    //语句中提前with出来的表
    private Set<String> withTables;

    private List<String> crmTables;

    private String insertTable;

    public static class CrmHiveTableCache {

        public static List<String> getTables() {
            String tables = CacheUtil.getValue("crm_hive_table");
            if (StringUtils.isNotBlank(tables)) {
                return JSON.parseArray(tables, String.class);
            }
            return new ArrayList<>();
        }

        public static void addTables(String table) {
            List<String> tables = getTables();
            if (!tables.contains(table)) {
                tables.add(table);
                CacheUtil.set("crm_hive_table", JSON.toJSONString(tables), 86400 * 365);
            }
        }
    }

    private HiveParser(String dbName, String sql) {
        this.dbName = dbName;
        this.sql = sql;
        this.withTables = new HashSet<>();
        this.crmTables = CrmHiveTableCache.getTables();
    }

    public void setInsertTable(String insertTable) {
        this.insertTable = insertTable;
        if (!crmTables.contains(insertTable)) {
            CrmHiveTableCache.addTables(insertTable);
        }
    }

    public static HiveParser build(String dbName, String sql) {
        return new HiveParser(dbName, sql);
    }

    /**
     * 执行解析动作
     */
    public String parse() {
        List<SQLStatement> sqlStatements = SQLUtils.parseStatements(sql, JdbcConstants.HIVE);
        ArgumentAssert.isTrue(sqlStatements.size() == 1);

        SQLStatement sqlStatement = sqlStatements.get(0);
        doParseStatement(sqlStatement);

        String resultSql = SQLUtils.toSQLString(sqlStatement, JdbcConstants.HIVE);
        return replaceFormat(resultSql);
    }

    private String replaceFormat(String sql) {
        //格式化逗号换行
        for (int i = 1; i < 10; i++) {
            StringBuilder regex = new StringBuilder("\n");
            for (int j = 0; j < i; j++) {
                regex.append("\t");
            }
            regex.append(", ");

            StringBuilder replacement = new StringBuilder(",\n");
            for (int j = 0; j < i; j++) {
                replacement.append("\t");
            }
            sql = sql.replaceAll(regex.toString(), replacement.toString());
        }

        //格式化ON
        for (int i = 1; i < 10; i++) {
            StringBuilder regex = new StringBuilder("\n");
            for (int j = 0; j < i; j++) {
                regex.append("\t");
            }
            regex.append("ON ");

            sql = sql.replaceAll(regex.toString(), " ON ");
        }

        //解决逗号到注释的问题
        Pattern p = Pattern.compile(" [-][-] .*,\n");
        Matcher m = p.matcher(sql);
        while (m.find()) {
            String group = m.group();
            String replaceGroup = ",";
            if (group.contains(",")) {
                if (group.contains("(") && group.contains("),")) {
                    group = group.split("[(]")[0];
                    replaceGroup += group;
                } else {
                    replaceGroup += group.replaceAll(",", "");
                }
                sql = sql.replaceAll(group, replaceGroup);
            }
        }
        sql = sql.replaceAll(",, --", ", --");

        //格式化缩进
        sql = sql.replaceAll("\tINNER JOIN", "INNER JOIN")
                .replaceAll("\tJOIN", "JOIN")
                .replaceAll("\tLEFT JOIN", "LEFT JOIN")
                .replaceAll("\tFULL JOIN", "FULL JOIN")
                .replaceAll("\tCROSS JOIN", "CROSS JOIN")
                .replaceAll("\tLEFT OUTER JOIN", "LEFT OUTER JOIN")
                .replaceAll("\tAND", "AND")
                .replaceAll("\tLATERAL VIEW", "LATERAL VIEW")
                .replaceAll("\tWHEN", " WHEN")
                .replaceAll("\tTHEN", " THEN")
                .replaceAll("\tELSE", " ELSE")
                .replaceAll("\tOR", "OR");

        return sql;
    }

    private void doParseStatement(SQLStatement statement) {
        if (statement instanceof HiveInsertStatement) {
            doParseInsert((HiveInsertStatement) statement);
        } else if (statement instanceof SQLSelectStatement) {
            doParseSelect(((SQLSelectStatement) statement).getSelect().getQueryBlock());
        }
    }

    private void doParseInsert(HiveInsertStatement statement) {
        //处理with
        doParseWith(statement.getWith());

        //处理分区
        List<SQLAssignItem> partitions = statement.getPartitions();
        if (partitions.size() == 1) {
            SQLAssignItem sqlAssignItem = partitions.get(0);
            SQLIdentifierExpr target = (SQLIdentifierExpr) sqlAssignItem.getTarget();
            if ("dayid".equals(target.getName())) {
                replaceTimeVariable((SQLCharExpr) sqlAssignItem.getValue());
            }
        }

        //处理表的库名
        SQLExpr expr = statement.getTableSource().getExpr();
        String tableName = null;
        if (expr instanceof SQLIdentifierExpr) {
            String name = ((SQLIdentifierExpr) expr).getName();
            String[] split = name.split("[.]");
            if (split.length == 1) {
                tableName = split[0];
            } else {
                tableName = split[1];
                ((SQLIdentifierExpr) expr).setName(split[1]);
            }
        } else if (expr instanceof SQLPropertyExpr) {
            tableName = ((SQLPropertyExpr) expr).getName();
            statement.getTableSource().setExpr(new SQLIdentifierExpr(tableName));
        }

        //记录新增的表
        setInsertTable(tableName);

        //处理select块
        doParseQuery(statement.getQuery().getQuery());
    }

    private void doParseWith(SQLWithSubqueryClause with) {
        if (with == null) {
            return;
        }
        List<SQLWithSubqueryClause.Entry> entries = with.getEntries();
        for (SQLWithSubqueryClause.Entry entry : entries) {
            withTables.add(entry.getAlias());
            doParseQuery(entry.getSubQuery().getQuery());
        }
    }

    /**
     * 替换 $v_date 成 ${v_date}
     */
    private void replaceTimeVariable(SQLCharExpr expr) {
        if (TIME_VARIABLES.contains(expr.getText())
                || TIME_VARIABLES.stream().anyMatch(v -> Pattern.compile(v).matcher(expr.getText()).find())) {
            String[] split = expr.getText().split("[$]");
            if (split.length == 2) {
                String text = "${" + split[1] + "}";
                expr.setText(text);
            }
        }
    }

    private void tableAddDb(SQLExprTableSource source) {
        SQLExpr expr = source.getExpr();
        if (expr instanceof SQLIdentifierExpr) {
            String name = ((SQLIdentifierExpr) expr).getName();
            String[] split = name.split("[.]");
            if (split.length == 1) {
                if (!withTables.contains(name) && !name.equals(insertTable)) {
                    name = (crmTables.contains(name) ? dbName : "ytdw") + "." + name;
                    ((SQLIdentifierExpr) expr).setName(name);
                }
            } else if (split[0].equals("rtdw")) {
                name = "ytdw." + split[1];
                ((SQLIdentifierExpr) expr).setName(name);
            }
        } else if (expr instanceof SQLPropertyExpr) {
            String ownerName = ((SQLPropertyExpr) expr).getOwnerName();
            if (ownerName.equals("rtdw")) {
                ((SQLPropertyExpr) expr).setOwner("ytdw");
            } else if (ownerName.equals("ytdw") && crmTables.contains(((SQLPropertyExpr) expr).getName())) {
                ((SQLPropertyExpr) expr).setOwner(dbName);
            }
        }
    }

    private void doParseQuery(SQLSelectQuery query) {
        if (query instanceof SQLSelectQueryBlock) {
            doParseSelect((SQLSelectQueryBlock) query);
        } else if (query instanceof SQLUnionQuery) {
            doParseUnion((SQLUnionQuery) query);
        }
    }

    private void doParseUnion(SQLUnionQuery query) {
        for (SQLSelectQuery relation : query.getRelations()) {
            doParseQuery(relation);
        }
    }

    private void doParseSelect(SQLSelectQueryBlock block) {
        //处理select
        block.getSelectList().forEach(s -> doParseExpr(s.getExpr()));

        //处理from
        doParseTableSource(block.getFrom());

        //处理where
        doParseExpr(block.getWhere());
    }

    private void doParseTableSource(SQLTableSource from) {
        if (from instanceof SQLExprTableSource) {
            tableAddDb((SQLExprTableSource) from);
        } else if (from instanceof SQLSubqueryTableSource) {
            doParseQuery(((SQLSubqueryTableSource) from).getSelect().getQuery());
        } else if (from instanceof SQLLateralViewTableSource) {
            doParseTableSource(((SQLLateralViewTableSource) from).getTableSource());
        } else if (from instanceof SQLJoinTableSource) {
            doParseTableSource(((SQLJoinTableSource) from).getLeft());
            doParseTableSource(((SQLJoinTableSource) from).getRight());
            doParseExpr(((SQLJoinTableSource) from).getCondition());
        } else if (from instanceof SQLUnionQueryTableSource) {
            doParseQuery(((SQLUnionQueryTableSource) from).getUnion());
        }
    }

    private void doParseExpr(SQLExpr expr) {
        if (expr == null) {
            return;
        }
        if (expr instanceof SQLBinaryOpExpr) {
            doParseExpr(((SQLBinaryOpExpr) expr).getLeft());
            doParseExpr(((SQLBinaryOpExpr) expr).getRight());
        } else if (expr instanceof SQLCharExpr) {
            replaceTimeVariable((SQLCharExpr) expr);
        } else if (expr instanceof SQLMethodInvokeExpr) {
            udfAddDb((SQLMethodInvokeExpr) expr);
            ((SQLMethodInvokeExpr) expr).getArguments().forEach(e -> doParseExpr(e));
        } else if (expr instanceof SQLCaseExpr) {
            ((SQLCaseExpr) expr).getItems().forEach(i -> doParseExpr(i.getConditionExpr()));
            doParseExpr(((SQLCaseExpr) expr).getElseExpr());
        }
    }

    private void udfAddDb(SQLMethodInvokeExpr expr) {
        if (expr.getOwner() != null) {
            return;
        }
        String methodName = expr.getMethodName();
        String[] split = methodName.split("[.]");
        if (split.length == 1) {
            if (YT_UDFS.contains(methodName)) {
                methodName = "ytdw." + methodName;
                expr.setMethodName(methodName);
            }
        }
    }
}
