查询模型

public class ChatMsgQuery extends CrmBaseQuery {

    private static final long serialVersionUID = -8613283515559437329L;

    /**
     * 企微实体id
     */
    private Long qwChatId;

    /**
     * 门店外部联系人userid
     */
    private String externalUserId;

    /**
     * 起始id，传了就表示查询该id后的记录（不包含当前id）
     */
    private Long beginId;

    /**
     * 结束id，传了就表示查询该id前的记录（不包含当前id）
     */
    private Long endId;

    /**
     * 消息开始时间
     */
    private Date beginTime;

    /**
     * 消息结束时间
     */
    private Date endTime;

    /**
     * 消息文案查询关键字
     */
    private String text;

}

消息模型

public class ChatMsgTO implements Serializable {

    private static final long serialVersionUID = -6931463015518735664L;

    /**
     * 消息id
     */
    private Long id;

    /**
     * 实体id
     */
    private Long chatId;

    /**
     * 实体名
     */
    private String chatName;

    /**
     * 头像
     */
    private String chatPhoto;

    /**
     * 单个记录的发言人
     * 1：右边，表示自己 （企微）
     * 2：左边，表示别人 （个微）
     * @see com.yt.touchc.api.chat.enums.ChatMsgDirectionEnum
     */
    private Integer direction;

    /**
     * 消息时间
     */
    private String msgTime;

    /**
     * 消息类型 text文本 image图片 voice语音
     * @see com.yt.touchc.api.chat.enums.ChatMsgTypeEnum
     */
    private String messageType;

    /**
     * 消息内容
     */
    private String content;
}


数据逻辑：

查询历史交接人员

SELECT * 
FROM t_crm_import_content 
WHERE import_type IN (4,5) 
AND result = '接替完毕' 
AND status = 2
AND json_extract(content_json, '$.externalUserId') = 'wm9D3QEAAAli9SHO1b7_Accm4LSyREYQ'
AND json_extract(content_json, '$.newQwUserId') = '13221026341'
AND create_time > '2025-06-16 00:00:00'
order by id desc
limit 1

查询聊天记录

SELECT * 
FROM t_crm_chat_msg_qw 
WHERE msg_type IN ('text', 'image', 'voice')
AND admin_chat_id = 21731682
AND shop_chat_id = 24151127
AND msg_time > '2025-12-16 00:00:00'
AND qw_action = 'send'
-- AND id > 987290719 下拉用
-- AND id < 987290719 定位到原始消息后上啦用
order by id desc
limit 10
