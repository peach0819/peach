企微在线和离线数据-crm库


select
t7.name as '大区名称',
t6.name as '组名',
t5.full_name as '销售名称',
t4.phone as '企微手机号',
t3.name as '企微昵称',
case t1.tuse_msg_type
when '902500002' then '在线'
else '离线'
end as '动作类型',
t1.tuse_msg_time as '时间',
t1.tuse_wx_id
from
crm.t_crm_tuse_qw_heartbeat_msg t1
left join (
select
qw_chat_id,
qw_tuse_id
from
crm.t_crm_chat_relation
group by
qw_chat_id,
qw_tuse_id
) t2 on t1.tuse_wx_id = t2.qw_tuse_id
left join crm.t_crm_chat t3 on t3.id = t2.qw_chat_id
left join crm.t_crm_work_phone t4 on t4.bind_qw_user_id = t3.qw_user_id
left join yt_ustone.t_user_admin t5 on t5.user_id = t4.bind_user_id
left join yt_ustone.t_department t6 on t6.id = t5.dept_id
left join yt_ustone.t_department t7 on t7.id = t6.parent_id

-- 去除注释就是最长的离线数据
-- where t1.is_valid=1
-- and t1.tuse_msg_type=902500005
order by t1.tuse_msg_time desc


触达率-离线库

with detail as (
select
shop_id,
admin_chat_id,
detail_status,
task_instance_id
from
yt_crm.ads_touch_task_detail
where
(
detail_status = 2
OR (
detail_status = 3
AND fail_type IN (3, 12)
)
)
and is_deleted = 0
),
instance as (
SELECT
id,
substr(start_time, 1, 8) as day
FROM
yt_crm.ads_touch_task_instance
WHERE
channel_type = 1
and is_deleted = 0
),
statistics as (
SELECT
instance.day,
detail.admin_chat_id,
count(distinct shop_id) as shop_count,
count(
distinct case
when detail_status = 2 then shop_id
else null
end
) as success_shop_count
FROM
detail
INNER JOIN instance ON detail.task_instance_id = instance.id
group by
instance.day,
detail.admin_chat_id
)
select
statistics.day `日期`,
parentDept.name `大区`,
dept.name `小组`,
charger.full_name `主管`,
admin.full_name `电销名`,
statistics.shop_count `目标触达门店数`,
statistics.success_shop_count `成功触达门店数`,
ROUND(statistics.success_shop_count * 100 / statistics.shop_count) `成功触达率`,
statistics.admin_chat_id,
chat.id chat_id,
chat.qw_user_id chat_qw_user_id,
workPhone.id work_phone_id,
workPhone.bind_user_id work_phone_bind_user_id
from
statistics
left join dwd_crm_chat_d chat on chat.id = statistics.admin_chat_id
and chat.dayid = '$v_date' and chat.is_deleted=0
left join dwd_crm_work_phone_d workPhone on workPhone.bind_qw_user_id = chat.qw_user_id
and workPhone.dayid = '$v_date' and workPhone.is_deleted=0
left join dwd_user_admin_d admin on admin.user_id = workPhone.bind_user_id
and admin.dayid = '$v_date' and admin.is_deleted=0
left join dwd_department_d dept on admin.dept_id = dept.id
and dept.dayid = '$v_date' and dept.is_deleted=0
left join dwd_department_d parentDept on parentDept.id = dept.parent_id
and parentDept.dayid = '$v_date' and parentDept.is_deleted=0
left join dwd_user_admin_d charger on charger.user_id = dept.charger
and charger.dayid = '$v_date' and charger.is_deleted=0
order by statistics.day desc
