3、基础信息查询服务

群信息查询

dubbo接口类：com.yt.touchc.api.chat.ChatGroupApi

groupType：对应枚举 com.yt.touchc.api.chat.enums.ChatGroupTypeEnum#WX

groupIds：关联com.yt.touchc.api.tuse.to.TuseWxGroupMsgTO#groupId

package com.yt.touchc.api.chat;

public interface ChatGroupApi {

    /**
     * 通过三方群id批量查询已入库数据
     *
     * @param groupType 群类型
     * @param groupIds  三方群id
     * @return 群列表
     */
    RpcResult<List<ChatGroupTO>> getByGroupIds(Integer groupType, List<String> groupIds);
}


群成员信息查询

dubbo接口类：com.yt.touchc.api.chat.ChatGroupMemberApi

chatGroupId：关联 com.yt.touchc.api.chat.to.ChatGroupTO#id

memberType：对应枚举 com.yt.touchc.api.chat.enums.ChatGroupMemberTypeEnum#WX_LINKER

groupUserIds：关联 com.yt.touchc.api.tuse.to.TuseWxGroupMsgTO#sendWxId 和 com.yt.touchc.api.tuse.to.TuseWxGroupMsgTO#toWxIds

package com.yt.touchc.api.chat;

public interface ChatGroupMemberApi {

    /**
     * 通过群id批量查询群成员列表
     *
     * @param chatGroupId  群id
     * @param memberType   群成员类型  com.yt.touchc.api.chat.enums.ChatGroupMemberTypeEnum
     * @param groupUserIds 群成员id列表
     * @return 群成员列表
     */
    RpcResult<List<ChatGroupMemberTO>> getByGroupUserIds(Long chatGroupId, Integer memberType, List<String> groupUserIds);
}
