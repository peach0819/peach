四. 数据库设计

ddl
create table sync_touch_vs_dkh_d (
id bigint(16) NOT NULL AUTO_INCREMENT,
creator varchar(32) NOT NULL COMMENT '创建人',
editor varchar(32) NOT NULL COMMENT '编辑人',
create_time datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
edit_time datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '修改时间',
is_deleted tinyint(4) COMMENT '是否删除:0-否 1-是',
dayid varchar(8) NOT NULL COMMENT '数仓表区块标示',
shop_id varchar(128) COMMENT '门店ID',
biz_value text COMMENT '大客户播报信息',
PRIMARY KEY (`id`),
KEY `idx_dayid_shop_id` (`dayid`, `shop_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='微商大客户播报回流表';

biz_value json结构

业务字段key业务字段value含义
shop_total_return门店总返点
t1_return出行爆品返点
t2_target出行品牌目标
t2_gmv出行品牌完成
t2_beyond出行品牌目标超出完成
t2_return出行品牌返点
t3_target母婴品牌目标
t3_gmv母婴品牌完成
t3_beyond母婴品牌目标超出完成
t3_return母婴品牌返点
max_return最高返点
