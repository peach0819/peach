########com.yangt.cif.dal.visitrecord.entity.VisitRecordDO
package com.yangt.cif.dal.visitrecord.entity;

import com.yt.crm.dal.crm.BaseDO;
import lombok.Data;

import java.util.Date;

@Data
public class VisitRecordDO extends BaseDO {
    /**
     * 自增主键
     */
    private Long id;
    /**
     * buid 默认0海拍客
     */
    private Integer buId;
    /**
     * 分销商id,0:海拍客,1:贝莱
     */
    private Integer dcId;
    /**
     * 小记id，全局唯一，作为其他表关联小记的外键
     */
    private Long recordId;
    /**
     * 拜访类别
     * <p>
     * 1：服务商经理拜访门店
     * 2：服务商经理拜访服务商
     * 3: BD拜访门店
     * 4: 电销拜访门店
     * 5: 服务商业务员拜访门店
     */
    private Integer recordType;
    /**
     * 出访人ID 销售账号
     */
    private String visitorId;
    /**
     * 出访人快照
     */
    private String visitorJson;
    /**
     * 陪访人id
     */
    private String withUserId;
    /**
     * 受访对象id，门店id/服务商id
     */
    private String objectId;
    /**
     * 受访对象快照
     */
    private String objectJson;
    /**
     * 拜访结果类型
     * 1为有效拜访
     * 2无效小记
     * 3为一般拜访
     */
    private Integer visitMode;
    /**
     * 拜访方式
     * <p>
     * 1、见面 2、电话 3、微信
     */
    private Integer visitType;
    /**
     * 拜访目的类型，多个逗号分隔
     * <p>
     * 各类小记自定义
     */
    private String visitAimsType;
    /**
     * 拜访状态
     * <p>
     * 0：未知；1：未完成；2：已完成
     */
    private Integer visitStatus;
    /**
     * 联系人id
     */
    private String linkerId;
    /**
     * 联系人快照
     */
    private String linkerJson;
    /**
     * 起始拜访时间
     */
    private Date visitTime;
    /**
     * 结束拜访时间
     */
    private Date endVisitTime;
    /**
     * 联系内容，长文本字段
     */
    private String content;
    /**
     * 拜访位置json，支持多个
     */
    private String positionJson;
    /**
     * 图片和语音等内容
     */
    private String fileJson;
    /**
     * 审核状态 0：未知 1：不合格；2:合格
     */
    private Integer verifyStatus;
    /**
     * 审核人
     */
    private String verifyUserId;
    /**
     * 审核时间
     */
    private Date verifyTime;
    /**
     * 拜访扩展的字段json格式内容
     */
    private String extraJson;
    /**
     * 通话记录code
     */
    private String conversationCode;
    /**
     * 联系计划id
     */
    private Long planId;
    /**
     * 下次联系计划id
     */
    private Long nextPlanId;
    /**
     * 项目id列表，逗号分隔
     */
    private String subjectIds;
}
